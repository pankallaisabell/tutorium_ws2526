    class Sortierbar {
        <<interface>>
        +vergleiche(v: Veranstaltung) int
    }

    class VeranstaltungsPlaner {
        -veranstaltungen: Veranstaltung[]
        +hauptMenue()
        #neuesEvent()
        #drucken()
    }

    class Veranstaltung {
        <<abstract>>
        -titel: String
        -ort: String
        +setTitel(t: String)
        +getTitel() String
        +getBeschreibung() String
    }

    class Messe {
        -datum: Datum
        +getTag() int
        +getMonat() int
    }

    class Konzert {
        -ticketStatus: Status
        -startZeit: DatumZeit
        +getStatus() Status
    }

    class Workshop {
        -datum: Datum
        -teilnehmer: Teilnehmer[]
    }

    class Teilnehmer {
        -name: String
        -nummer: int
        -gebDatum: Datum
    }

    class Status {
        <<enum>>
        AUSVERKAUFT
        VERFUEGBAR
        WARTELISTE
    }

    class Datum {
        -tag: int
        -monat: int
        -jahr: int
        +Datum(t: int, m: int, j: int)
        +setTag(t: int)
        +getTag() int
        ...
    }

    class Zeit {
        -stunde: int
        -minute: int
        +Zeit(h: int, m: int)
        ...
    }

    class DatumZeit {
        -datum: Datum
        -zeit: Zeit
        +DatumZeit(d: Datum, z: Zeit)
        ...
    }
